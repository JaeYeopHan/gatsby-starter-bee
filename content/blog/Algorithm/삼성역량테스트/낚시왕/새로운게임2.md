자세한 문제는 [낚시왕 17143](https://www.acmicpc.net/problem/17143)를 참고
<br/><br/>

나머지 연산을 이용하여 시간을 줄이는 것이 핵심인 문제였다.

<br/><br/>

```c
#include<iostream>
#include<cstring>
#include<vector>
using namespace std;

int ret = 0, R, C, m;
int arr[100][100];
int dr[4] = { -1,1,0,0 };
int dc[4] = { 0,0,1,-1 };

//상어를 담을 구조체
struct SHARK {
	int r, c, s, d, z;
}sh;

//구조체 벡터
vector<SHARK> v;

int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
	cin >> R >> C >> m;

	//맵 arr을 0으로 초기화
	memset(arr, -1, sizeof(arr));

	//나머지 연산에 사용할 mr, mc
	int mr = (R - 1) * 2;
	int mc = (C - 1) * 2;

	//상어의 정보를 구조체에 저장
	for (int i = 0; i < m; ++i) {
		cin >> sh.r >> sh.c >> sh.s >> sh.d >> sh.z;
		sh.r--; sh.c--; sh.d--;

		//속도를 mr, mc보다 작게 설정(속도 향상)
		if (sh.d == 0 || sh.d == 1)sh.s = sh.s%mr;
		else if (sh.d == 2 || sh.d == 3)sh.s = sh.s%mc;
		v.push_back(sh);

		//맵에 물고기의 인덱스를 저장
		arr[sh.r][sh.c] = i;
	}
	//낚시왕 이동
	for (int i = 0; i < C; ++i) {
		//낚시 행동
		for (int j = 0; j < R; ++j) {
			//물고기가 있으면(-1이 아니면) 낚시
			if (arr[j][i] != -1) {
				ret += v[arr[j][i]].z;
				v[arr[j][i]].d = -1;
				arr[j][i] = -1;
				break;
			}
		}
		//낚시 후 맵 -1로 초기화
		memset(arr, -1, sizeof(arr));
		for (int j = 0; j < m; ++j) {
			if (v[j].d == -1)continue;
			int r = v[j].r;
			int c = v[j].c;
			int d = v[j].d;
			int s = v[j].s;
			int z = v[j].z;
			//물고기 s(속력)만큼 이동
			while (s--) {
				int nr = r + dr[d];
				int nc = c + dc[d];
				if (nr < 0 || nc < 0 || nr >= R || nc >= C) {
					if (d == 0) d = 1;
					else if (d == 1) d = 0;
					else if (d == 2) d = 3;
					else if (d == 3) d = 2;
				}
				r += dr[d];
				c += dc[d];
			}
			v[j].r = r; v[j].c = c; v[j].d = d;
			if (arr[r][c] != -1) {
				if (v[arr[r][c]].z < z) {
					//먹힌 물고기의 방향을 -1로 만들어 없는셈 친다
					v[arr[r][c]].d = -1;
					arr[r][c] = j;
				}
				//먹힌 물고기의 방향을 -1로 만들어 없는셈 친다
				else v[j].d = -1;
			}
			else arr[r][c] = j;
		}
	}
	cout << ret;
	return 0;
}
```
