자세한 문제는 [연구소 14502](https://www.acmicpc.net/problem/14502)를 참고하자
<br/><br/><br/><br/>

먼저 완전탐색으로 접근한다. 삼성은 A형 시험 뿐만 아니라 sw역량테스트에서도 완전탐색으로 접근하는 문제가 많다.


```c
//depth = 깊이, r = row, c = column
void dfs(int depth, int r, int c) {
    //문제에서 벽을 3개만세우므로, 깊이가 3이면 retrun해서 경우의 수를 줄여줍니다
	if (depth == 3) {
        //벽을 3개 세웠을 때, 바이러스를 퍼트리는 함수로 이동합니다.
		bfs();
		return;
	}
    //두 개의 for문을 통해 모든 맵을 탐색합니다.
	int map_temp[8][8];
	for (int i = r; i < n; ++i) {
		for (int j = c; j < m; ++j) {
            //if문으로 경우의 수를 줄여주는 게 좋습니다.
			if (arr[i][j] != 0)continue;
            //현재 맵을 복사합니다.
			map_cp(map_temp, arr);
            //벽을 세웁니다
			arr[i][j] = 1;
            //깊이를 +1, 현재위치에서 다시 탐색합니다.  
			dfs(depth + 1, i, j);
            //복사했던 맵을 가져옵니다
			map_cp(arr, map_temp);
            //벽을 제거합니다.
			arr[i][j] = 0;
		}
		//c=0해야 열의 처음부터 탐색함
		c = 0;
	}
}
```

dfs를 통해 기둥 세개를 세웠고,  bfs를 통해 바이러스를 퍼트려야 한다. 

```c
void bfs() {
    //q에 위치 저장한 바이러스 넣어줌
	for (int i = 0; i < v.size(); ++i)
		q.push({ v[i].first, v[i].second });
	//bfs
	while (!q.empty()) {
        //현재 위치
		int r = q.front().first;
		int c = q.front().second;
        //pop을 안하면 무한루프에 빠집니다
		q.pop();
        //4방향 탐색
		for (int i = 0; i < 4; ++i) {
            //nextr,nextc를 뜻하며 다음 칸의 값을 갖습니다.
			int nr = r + dr[i];
			int nc = c + dc[i];
            //맵을 벗어나지 않게 조건을 추가합니다, 벽이어도 이동하지 못합니다.
			if (nr < 0 || nc < 0 || nr >= n || nc >= m || arr[nr][nc] != 0)continue;
            //바이러스=2
			arr[nr][nc] = 2;
			q.push(make_pair(nr, nc));
		}
	}
    //바이러스 없는 지역 세서 cnt에 저장
	int cnt = 0;
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < m; ++j)
			if (arr[i][j] == 0)cnt++;

	if (ret < cnt) {
		ret = cnt;
	}
}
```

```c
//백준 14502 연구소 191226
//keypoint : 백트래킹, 맵 저장, 조합
#include <iostream>
#include <queue>
#include <vector>
using namespace std;

int n, m, ret = 0;
int arr[8][8];
//4방향 탐색할 변수
int dr[4] = { 0,0,1,-1 };
int dc[4] = { 1,-1,0,0, };
queue<pair<int, int> > q;
vector<pair<int, int> > v;

//맵 카피함수
void map_cp(int arr1[8][8], int arr2[8][8]) {
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < m; ++j)
			arr1[i][j] = arr2[i][j];
}

//바이러스 퍼트리기
void bfs() {
	for (int i = 0; i < v.size(); ++i)
		q.push({ v[i].first, v[i].second });

	while (!q.empty()) {
		int r = q.front().first;
		int c = q.front().second;
		q.pop();
		for (int i = 0; i < 4; ++i) {
			int nr = r + dr[i];
			int nc = c + dc[i];
			if (nr < 0 || nc < 0 || nr >= n || nc >= m || arr[nr][nc] != 0)continue;
			arr[nr][nc] = 2;
			q.push(make_pair(nr, nc));
		}
	}
	int cnt = 0;
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < m; ++j)
			if (arr[i][j] == 0)cnt++;

	if (ret < cnt) {
		ret = cnt;
	}
}

void dfs(int depth, int r, int c) {
	if (depth == 3) {
		bfs();
		return;
	}
	//기둥세우고 맵 저장,불러오기
	int map_temp[8][8];
	for (int i = r; i < n; ++i) {
		for (int j = c; j < m; ++j) {
			if (arr[i][j] != 0)continue;
			map_cp(map_temp, arr);
			arr[i][j] = 1;
			dfs(depth + 1, i, j);
			map_cp(arr, map_temp);
			arr[i][j] = 0;
		}
		//c=0해야 열의 처음부터 탐색함
		c = 0;
	}
}

int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
	cin >> n >> m;
	//맵 그리고 바이러스 v에 push
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < m; ++j) {
			cin >> arr[i][j];
			if (arr[i][j] == 2)v.push_back({ i,j });
		}
	}

	dfs(0, 0, 0);
	cout << ret;
	return 0;
}
```
